  // TASK TOGGLE — tap expands context, checkbox checks
  function toggleTask(item) {
    const cb = item.querySelector('.task-cb');
    const text = item.querySelector('.task-text');
    cb.classList.toggle('checked');
    text.classList.toggle('done');
    item.classList.toggle('expanded');
    saveState();
  }

  // PERSIST
  function saveState() {
    const items = document.querySelectorAll('.task-item');
    const state = Array.from(items).map(item => ({
      checked: item.querySelector('.task-cb').classList.contains('checked'),
      expanded: item.classList.contains('expanded')
    }));
    try { localStorage.setItem('lloyd_cmd_v2', JSON.stringify(state)); } catch(e) {}
  }

  function loadState() {
    try {
      const saved = JSON.parse(localStorage.getItem('lloyd_cmd_v2') || '[]');
      const items = document.querySelectorAll('.task-item');
      items.forEach((item, i) => {
        if (saved[i]) {
          if (saved[i].checked) {
            item.querySelector('.task-cb').classList.add('checked');
            item.querySelector('.task-text').classList.add('done');
          }
          if (saved[i].expanded) item.classList.add('expanded');
        }
      });
    } catch(e) {}
  }

  function resetAll() {
    if (!confirm('Reset all daily tasks?')) return;
    document.querySelectorAll('.task-cb').forEach(cb => cb.classList.remove('checked'));
    document.querySelectorAll('.task-text').forEach(t => t.classList.remove('done'));
    document.querySelectorAll('.task-item').forEach(i => i.classList.remove('expanded'));
    try { localStorage.removeItem('lloyd_cmd_v2'); } catch(e) {}
  }

  // COPY SUMMARY
  function copyProgress() {
    const platforms = ['PagoSeguro','Mahali','PagoYa'];
    const cards = document.querySelectorAll('.platform-card');
    let summary = `LLOYD CMD CENTER — ${new Date().toDateString()}\n\n`;
    cards.forEach((card, i) => {
      const tasks = card.querySelectorAll('.task-item');
      const done = card.querySelectorAll('.task-cb.checked').length;
      summary += `${platforms[i]}: ${done}/${tasks.length} tasks done\n`;
      tasks.forEach(task => {
        const checked = task.querySelector('.task-cb').classList.contains('checked') ? '✓' : '○';
        summary += `  ${checked} ${task.querySelector('.task-text').textContent.trim()}\n`;
      });
      summary += '\n';
    });
    try {
      navigator.clipboard.writeText(summary);
      alert('Progress copied to clipboard!');
    } catch(e) { alert(summary); }
  }

  // PWA INSTALL BANNER (show on mobile if not installed)
  function showInstallBanner() {
    const isStandalone = window.navigator.standalone || window.matchMedia('(display-mode: standalone)').matches;
    const dismissed = localStorage.getItem('banner_dismissed');
    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
    if (isMobile && !isStandalone && !dismissed) {
      document.getElementById('installBanner').style.display = 'flex';
    }
  }

  function dismissBanner() {
    document.getElementById('installBanner').style.display = 'none';
    try { localStorage.setItem('banner_dismissed', '1'); } catch(e) {}
  }

  loadState();
  showInstallBanner();
